## Getting Started

```bash
$ yarn add @stackflow/core
```

```tsx
import React from 'react'
import ReactDOM from 'react-dom'

import { stackflow } from '@stackflow/core';

const { Stack, useFlow } = stackflow({
  // ...
});

const App: React.FC = () => {
  return (
    <Stack />
  );
};

ReactDOM.render(<App />, ...)
```
